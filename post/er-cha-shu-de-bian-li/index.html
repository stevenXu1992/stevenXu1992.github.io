public static int getMaxWidth(Node head){
  if(head ==null){
    return 0;
  }
  int maxWidth = 0;
  int curWidth = 0;
  int curLevel = 0;
  HashMap<Node, Integer> levelMap = new HashMap<>();
  LinkedList<Node> queue = new LinkedList<>();
  levelMap.put(head, 1);
  queue.add(head);
  Node cur = null;
  Node left = null;
  Node right = null;
  while(!queue.isEmpty()){
    cur = queue.poll();
    left = cur.left;
    right = cur.right;
    if(left != null){
      levelMap.put(left, levelMap.get(cur) + 1);
      queue.add(left);
    }
    if(right != null){
      levelMap.put(right, levelMap.get(cur) + 1);
      queue.add(right);
    }
    if(levelMap.get(cur) > curLevel){
      curWidth = 0;
      curLevel = levelMap.get(cur);
    }else{
      curWidth++;
    }
    maxWidth = Math.max(curWidth, maxWidth);
  }
  return maxWidth;
}
