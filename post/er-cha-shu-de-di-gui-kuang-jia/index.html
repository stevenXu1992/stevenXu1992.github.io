public static boolean searchTree(Node head){
  if(head == null){
    return true;
  }
  Stack<Node> stack = new Stack<>();
  Node cur = head;
  int pre = Integer.MIN_VALUE;
  while(!stack.isEmpty() || cur != null){
    if(cur != null){
      stack.push(cur);
      cur = cur.left;
    }else{
      cur = stack.pop();
      if(cur.data <= pre){
        return false;
      }
      pre = cur.data;
      cur = cur.right;
    }
  }
  return true;
}
