图的经典算法

2020-05-18 Views 图693字5 min read繁

1. 模版

public class Graph{
  public HashMap<Integer, Node> nodes;
  public HashSet<Edge> edges;
  public Graph(){
    nodes = new HashMap<>();
    edges = new HashSet<>();
  }
}

public class Node{
  public int value;
  public int in;
  public int out;
  public ArrayList<Node> nexts;
  public ArrayList<Edge> edges;
  public Node(int data){
    this.value = data;
    in = 0;
    out = 0;
    nexts = new ArrayList<>();
    edges = new ArrayList<>();
  }
}

public class Edge{
  public int weight;
  public Node from;
  public Node to;
  public Edge(int weight, Node from, Node to){
    this.weight = weight;
    this.from = from;
    this.to = to;
  }
}

2. 图的遍历

2.1 宽度优先遍历

public static void BFS(Node node){
  if(node == null){
    return;
  }
  LinkedList<Node> queue = new LinkedList<>();
  HashSet<Node> set = new HashSet<>();
  queue.add(node);
  set.add(node);
  while(!queue.isEmpty()){
    Node cur = queue.poll();
    //具体遍历操作
    for(Node n : cur.nexts){
      if(!set.contains(n)){
        queue.add(n);
        set.add(n);
      }
    }
  }
}

2.2 深度优先遍历

public static void DFS(Node node){
  if(node == null){
    return;
  }
  Stack<Node> stack = new Stack<>();
  HashSet<Node> set = new HashSet<>();
  stack.push(node);
  set.add(node);
  //具体遍历操作
  while(!stack.isEmpty()){
    Node cur = stack.pop();
    for(Node n : cur.nexts){
      if(!set.contains(n)){
        stack.push(cur);
        stack.push(n);
        set.add(n);
        //具体遍历操作
        break;
      }
    }
  }
}

3.拓扑排序

public static List<Node> topologySort(Graph graph){
  HashMap<Node, Integer> map = new HashMap<>();
  LinkedList<Node> queue = new LinkedList<>();
  ArrayList<Node> list = new ArrayList<>();
  
  for(Node n : graph.nodes.values()){
    map.put(n, n.in);
    if(n.in == 0){
      queue.add(n);
    }
  }
  
  while(!queue.isEmpty()){
    Node cur = queue.pop();
    list.add(cur);
    for(Node n : cur.nexts){
      map.put(n, map.get(n) - 1);
      if(map.get(n) == 0){
        queue.add(n);
      }
    }
  }
  return list;
}

4. Prim算法

public static class EdgeComparator implements Comparator<Edge>{
  public int compare(Edge o1, Edge o2){
    //自定义比较器:
    // o1 - o2 < 0 : o1 --> o2
    // o2 - o1 < 0 : o2 --> o1       
    return o1.weight - o2.weight;
  }
}

public static Set<Edge> prim(Graph graph){
  PriorityQueue<Edge> queue = new PriorityQueue<>();
  HashSet<Node> set = new HashSet<Node>();
  HashSet<Edge> res = new HashSet<Edge>();
  
  for(Node node : graph.nodes.values()){
    if(!set.contains(node)){
      set.add(node);
      for(Edge edge : node.edges){
        queue.add(edge);
      }
      while(!queue.isEmpty()){
        Edge edge = queue.poll();
        if(!set.contains(edge.to)){
          res.add(edge);
          set.add(edge.to);
          for(Edge e : edge.to.edges){
            queue.add(e);
          }
        }
      }
    }
  }
  return res;
}

5. Dijkstra算法

public static HashMap<Node, Integer> dijkstra(Node head){
  
  //distanceMap: 存储出发点至其他各点的距离
  HashMap<Node, Integer> distanceMap = new HashMap<>();
  //lockedSet: 用于记录已锁定的点
  HashSet<Node> lockedSet = new HashSet<>();
  distanceMap.put(head, 0);
  //unlockedMinNode: 找出distanceMap中距离最小且未锁定的点
  Node minNode = unlockedMinNode(distanceMap, lockedSet);
  
  while(minNode != null){
    int dis = distanceMap.get(minNode);
    for(Edge edge : minNode.edges){
      if(!distanceMap.containsKey(edge.to)){
        distanceMap.put(edge.to, dis + edge.weight);
      }
      //若distanceMap中已存在，则比较原距离和新发现的边所更新的距离
      distanceMap.put(edge.to, 
                      Math.min(distanceMap.get(edge.to), 
                      				 dis + edge.weight));
    }
    lockedSet.add(minNode);
    minNode = unlockedMinNode(distanceMap, lockedSet);
  }
  return distanceMap;
}

public static Node unlockedMinNode(
  HashMap<Node, Integer> disMap, HashSet<Node> lockedSet){
  Node minNode = null;
  int minDis = Integer.MAX_VALUE;
  for(Map.Entry<Node, Integer> entry : disMap.entrySet()){
    Node n = entry.getKey();
    int dis = entry.getValue();
    if(!lockedSet.contains(n) && dis < minDis){
      minNode = n;
      minDis = dis;
    }
  }
  return minNode;
}
