public static void BFS(Node node){
  if(node == null){
    return;
  }
  LinkedList<Node> queue = new LinkedList<>();
  HashSet<Node> set = new HashSet<>();
  queue.add(node);
  set.add(node);
  while(!queue.isEmpty()){
    Node cur = queue.poll();
    //å…·ä½“éå†æ“ä½œ
    for(Node n : cur.nexts){
      if(!set.contains(n)){
        queue.add(n);
        set.add(n);
      }
    }
  }
}

public static List<Node> topologySort(Graph graph){
  HashMap<Node, Integer> map = new HashMap<>();
  LinkedList<Node> queue = new LinkedList<>();
  ArrayList<Node> list = new ArrayList<>();
  
  for(Node n : graph.nodes.values()){
    map.put(n, n.in);
    if(n.in == 0){
      queue.add(n);
    }
  }
  
  while(!queue.isEmpty()){
    Node cur = queue.pop();
    list.add(cur);
    for(Node n : cur.nexts){
      map.put(n, map.get(n) - 1);
      if(map.get(n) == 0){
        queue.add(n);
      }
    }
  }
  return list;
}

public static Set<Edge> prim(Graph graph){
  PriorityQueue<Edge> queue = new PriorityQueue<>();
  HashSet<Node> set = new HashSet<Node>();
  HashSet<Edge> res = new HashSet<Edge>();
  
  for(Node node : graph.nodes.values()){
    if(!set.contains(node)){
      set.add(node);
      for(Edge edge : node.edges){
        queue.add(edge);
      }
      while(!queue.isEmpty()){
        Edge edge = queue.poll();
        if(!set.contains(edge.to)){
          res.add(edge);
          set.add(edge.to);
          for(Edge e : edge.to.edges){
            queue.add(e);
          }
        }
      }
    }
  }
  return res;
}

public static Node unlockedMinNode(
  HashMap<Node, Integer> disMap, HashSet<Node> lockedSet){
  Node minNode = null;
  int minDis = Integer.MAX_VALUE;
  for(Map.Entry<Node, Integer> entry : disMap.entrySet()){
    Node n = entry.getKey();
    int dis = entry.getValue();
    if(!lockedSet.contains(n) && dis < minDis){
      minNode = n;
      minDis = dis;
    }
  }
  return minNode;
}
