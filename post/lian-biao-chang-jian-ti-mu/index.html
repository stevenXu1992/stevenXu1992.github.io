public static boolean palindrome1(Node head){
  Stack<Node> stack = new Stack<Node>();
  Node cur = head;
  while(cur != null){
    stack.push(cur);
    cur = cur.next;
  }
  while(!stack.isEmpty()){
    Node n = stack.pop();
    if(head.value != n.value){
      return false;
    }
    head = head.next;
  }
  return true;
}

  if(head == null || head.next == null 
     || head.next.next == null){
    return null;
  }
  Node slow = head.next;
  Node fast = head.next.next;
  while(slow != fast){
    slow = slow.next;
    fast = fast.next.next;
  }
  fast = head;
  while(slow != fast){
    slow = slow.next;
    fast = fast.next;
  }
  return fast;
}
